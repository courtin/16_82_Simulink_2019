1	function [CX,CL,CY,Cl,Cm,Cn]	=	AeroModelSSTOL(x,u,Mach,alphar,betar,V,airplane)
  2%	SSTOL Aero Model
  3%   x(1) = u_b          Body axis u velocity, m/s
  4%   x(2) = w_b          Body axis w velocity, m/s
  5%   x(3) = q            Pitch rate, rad/s
  6%   x(4) = theta        Pitch euler angle, rad
  7%   x(5) = v_b          Body axis v velocity, m/s
  8%   x(6) = p            Roll rate, rad/s
  9%   x(7) = r            Yaw rate, rad/s
 10%   x(8) = phi          Roll euler angle, rad
 11%   x(9) = psi          Yaw euler angle, rad
 12%   x(10) = xe          North position, Earth frame, m
 13%   x(11) = ye          East position, Earth frame, m
 14%   x(12) = ze =-h      Negative of altitude, Earth frame, m
 15
 16%   u(1) = elevator deflection, rad, positive trailing edge down
 17%   u(2) = rudder deflection, rad, positive trailing edge down
 18%   u(3) = aileron deflection, rad, positive left trailing edge down
 19%   u(4) = left flap deflection, rad, positive trailing edge down
 20%   u(5) = right flap deflection, rad, positive trailing edge down
 21%   u(6:end) = u_t, thrust control vector, 0-1
 22
 23%   u_thrust(1) = left blowing throttle, 0-1
 24%   u_thrust(2) = right blowing throttle, 0-1
 25%   u_thrust(3) = left thruster throttle, 0-1
 26%   u_thrust(4) = right thruster throttle, 0-1
 27
 28%	Inertial, Geometric, and Aerodynamic Properties
 29%   *** STOL_Input.m must first be run to save Airplane.mat***
 30	A=load('Airplane.mat');
 31    airplane = A.airplane;
 32    
 33
 34    %Handy variable shorthands
 35    i_w = airplane.geometry.Wing.i;
 36    i_t = airplane.geometry.Htail.i;
 37    cbar = airplane.geometry.Wing.cbar;
 38    e = airplane.aero.Wing.e; %TODO: Make this a function of delta_f, etc.
 39    AR = airplane.geometry.Wing.AR;
 40    clat = airplane.aero.Htail.cla;
 41    Sh = airplane.geometry.Htail.S;
 42    Sv = airplane.geometry.Vtail.S;
 43    Sw = airplane.geometry.Wing.S;
 44    lh = airplane.geometry.Htail.l;
 45    lv = airplane.geometry.Vtail.l;
 46    b  = airplane.geometry.Wing.b;
 47    Vv = Sv/Sw*lv/b;
 48    Vh = Sh/Sw*lh/cbar;
 49    tau_h = airplane.aero.Htail.tau;
 50    x_cg = airplane.weights.xcg;
 51    x_acw = airplane.aero.Wing.x_ac;
 52    
 53    alt = -x(12);
 54    
 55    
 56    alpha_w     = alphar + i_w;
 57	a_w_deg	=	rad2deg(alpha_w);
 58    
 59    %Thrust Control vector
 60    u_t = u(6:end);
 61    d_BL = u_t(1);
 62    d_BR = u_t(2);
 63    d_TL = u_t(3);
 64    d_TR = u_t(4);
 65
 66%   Blowing and Thrust Calculations
 67%   ====================================
 68    [dCJ_BL, ~, CT_BL] = propulsor_perf(d_BL, airplane.propulsion.left_blower, cbar, alt, V);
 69    [dCJ_BR, ~, CT_BR] = propulsor_perf(d_BR, airplane.propulsion.right_blower, cbar, alt, V);
 70    [~, T_L, CT_CL] = propulsor_perf(d_TL, airplane.propulsion.left_cruiser, cbar, alt, V);
 71    [~, T_R, CT_CR] = propulsor_perf(d_TR, airplane.propulsion.right_cruiser, cbar, alt, V);
 72
 73%   Stability Derivates from JVL
 74%   ====================================
 75%   Currently these are just keyed off of the left flap deflection; no
 76%   provision yet for these as a result of asymmetric flap deflections
 77
 78    [CLde, CLq, ...
 79    Cmde, Cmq, ...
 80    CYb, CYdA, CYdR, CYr, CYp,...
 81    Clb, CldA, CldR, Clr, Clp,...
 82    Cnb, CndA, CndR, Cnr, Cnp] = get_JVL_derivatives(u(4), dCJ_BL, airplane);
 83
 84
 85%	CL Calculations 
 86%	====================================
 87
 88    %Wing Lift Coefficient
 89    %C_L_wing
 90    %Currently this interpolator is only for 40deg flaps; any
 91    %flap changes are ignored except the zero deg flap case.
 92    
 93    flap_L_deg = round(rad2deg(u(4)),0);
 94    flap_R_deg = round(rad2deg(u(5)),0);
 95    
 96    cl_left = getCLwing(a_w_deg,dCJ_BL,flap_L_deg,airplane);
 97    cl_right = getCLwing(a_w_deg,dCJ_BR,flap_R_deg,airplane);
 98    
 99    CLw = .9*(cl_left + cl_right)/2;
100    
101    %Tail Lift Coefficient
102    %C_L_ht
103    eta_h = 1; %Update with better method of estimating tail dynamic pressure ratio
104    eps = 2*CLw/(pi*e*AR); %May want to use M&S method here instead
105    a_h = (alphar+i_t-eps);
106    
107    eta = clat/(2*pi);
108    sweep_h = airplane.geometry.Htail.sweep;
109    CLah = (2*pi*AR)/(2+sqrt(4*(AR/eta)^2*(1+tan(sweep_h)^2)));
110    CLt = CLah*a_h*Sh/Sw*eta_h;
111    %Elevator and Pitch Rate effects
112    %CLde = airplane.stability.CLde;
113    CLde_check = Sh/Sw*CLah*tau_h;
114    %CLde = CLde_check; %Very different drom JVL; need to debug
115    %CLq = airplane.stability.CLq;
116    CLq_check = -2*Vh*CLah*eta_h;
117    %CLq = CLq_check; %Very different from JVL; need to debug
118    CL = CLw + CLah*a_h + CLq*x(3)*(cbar/(2*V)) + CLde*u(1);
119
120%	CX Calculations 
121%	====================================
122    cx_left = getCXwing(a_w_deg,dCJ_BL,flap_L_deg,airplane);
123    cx_right = getCXwing(a_w_deg,dCJ_BR,flap_R_deg,airplane);
124    
125    CXw = (cx_left+cx_right)/2;
126    CDi = CLw^2/(pi*AR*e+2*(CT_BL+CT_BR));
127    CDp = .02;  %Placeholder, this has little effect on the high-lift cases
128    
129    CX = CXw + CDi + CDp - CT_CL - CT_CR;
130
131%	Cm Calculations 
132%	====================================    
133    cm_left = getCMwing(a_w_deg,dCJ_BL,flap_L_deg,airplane);
134    cm_right = getCMwing(a_w_deg,dCJ_BR,flap_R_deg,airplane);
135    
136    
137    Cmw = (cm_left + cm_right)/2;
138    Cmh = -Vh*eta_h*CLah*a_h;
139    
140    %Cmde = airplane.stability.Cmde;
141    %Cmq = airplane.stability.Cmq;
142    
143    Cmaf = 0; %Negelect fuselage contributions for now
144    Cmf = Cmaf*alphar;
145    
146    Cm = CLw*(x_cg-x_acw)/cbar + Cmw + Cmh + Cmf + Cmde*u(1) + Cmq*x(3)*(cbar/(2*V));
147    
148    	
149%	Current Lateral-Directional Characteristics
150%	===========================================
151%	Cl Calculations 
152%	==================================== 
153%	Rolling Moment Coefficient
154
155	%Cl      =	(ClBr*betar + CldRr*u(2)) + Clrr * x(7) + Clpr * x(6) ...
156    %            + (CldAr*u(3));%
157									% Total Rolling-Moment Coefficient
158    Cl = 0;
159
160%	CY Calculations 
161%	==================================== 
162%	Side-Force Coefficient
163	%CY	=	(CYBr*betar + CYdRr*u(2)) + (CYdAr*u(3));% + CYdASr*u(5));
164									% Total Side-Force Coefficient
165    CY = 0;
166%	Cn Calculations 
167%	====================================                                     
168%	Yawing Moment Coefficient
169	%Cn	=	(CnBr*betar + CndRr*u(2)) + Cnrr * x(7) + Cnpr * x(6) ...
170	%		+ (CndAr*u(3));% + CndASr*u(5));
171									% Total Yawing-Moment Coefficient
172    Cn = 0;
173
