1	function [CX,CL,CY,Cl,Cm,Cn]	=	AeroModelSSTOL(x,u,Mach,alphar,betar,V)
  2%	SSTOL Aero Model
  3%   x(1) = u_b          Body axis u velocity, m/s
  4%   x(2) = w_b          Body axis w velocity, m/s
  5%   x(3) = q            Pitch rate, rad/s
  6%   x(4) = theta        Pitch euler angle, rad
  7%   x(5) = v_b          Body axis v velocity, m/s
  8%   x(6) = p            Roll rate, rad/s
  9%   x(7) = r            Yaw rate, rad/s
 10%   x(8) = phi          Roll euler angle, rad
 11%   x(9) = psi          Yaw euler angle, rad
 12%   x(10) = xe          North position, Earth frame, m
 13%   x(11) = ye          East position, Earth frame, m
 14%   x(12) = ze =-h      Negative of altitude, Earth frame, m
 15
 16%   u(1) = elevator deflection, rad, positive trailing edge down
 17%   u(2) = rudder deflection, rad, positive trailing edge down
 18%   u(3) = aileron deflection, rad, positive left trailing edge down
 19%   u(4) = left flap deflection, rad, positive trailing edge down
 20%   u(5) = right flap deflection, rad, positive trailing edge down
 21%   u(6:end) = u_t, thrust control vector, 0-1
 22
 23%   u_thrust(1) = left blowing throttle, 0-1
 24%   u_thrust(2) = right blowing throttle, 0-1
 25%   u_thrust(3) = left thruster throttle, 0-1
 26%   u_thrust(4) = right thruster throttle, 0-1
 27
 28%	Inertial, Geometric, and Aerodynamic Properties
 29%   *** STOL_Input.m must first be run to save Airplane.mat***
 30	
 31
 32    A=load('./SSTOL_Aero_Model/Airplane.mat');
 33    airplane=A.airplane;
 34    
 35    %cl_fit40 = load('./SSTOL_Aero_Model/cl_fit40.mat');
 36    
 37    %fits40 = load('./SSTOL_Aero_Model/40df_fits.mat');
 38    %cl_fit = fits40.cl_fit;
 39    %cm_fit = fits40.cm_fit;
 40    %cx_fit = fits40.cx_fit;
 41    
 42    
 43
 44    
 45    %Handy variable shorthands
 46    i_w = airplane.geometry.Wing.i;
 47    i_t = airplane.geometry.Htail.i;
 48    cbar = airplane.geometry.Wing.cbar;
 49    e = airplane.aero.Wing.e; %TODO: Make this a function of delta_f, etc.
 50    AR = airplane.geometry.Wing.AR;
 51    clat = airplane.aero.Htail.cla;
 52    Sh = airplane.geometry.Htail.S;
 53    Sv = airplane.geometry.Vtail.S;
 54    Sw = airplane.geometry.Wing.S;
 55    lh = airplane.geometry.Htail.l;
 56    lv = airplane.geometry.Vtail.l;
 57    b  = airplane.geometry.Wing.b;
 58    Vv = Sv/Sw*lv/b;
 59    Vh = Sh/Sw*lh/cbar;
 60    tau_h = airplane.aero.Htail.tau;
 61    x_cg = airplane.weights.xcg;
 62    x_acw = airplane.aero.Wing.x_ac;
 63    
 64    alt = -x(12);
 65    
 66    rad2deg = 180/pi;
 67    
 68    alpha_w     = alphar + i_w;
 69	a_w_deg	=	(alpha_w)*rad2deg;
 70    
 71    %Thrust Control vector
 72    u_t = u(6:end);
 73    d_BL = u_t(1);
 74    d_BR = u_t(2);
 75    d_TL = u_t(3);
 76    d_TR = u_t(4);
 77
 78%   Blowing and Thrust Calculations
 79%   ====================================
 80    [dCJ_BL, ~, CT_BL] = propulsor_perf(d_BL, airplane.propulsion.left_blower, cbar, alt, V);
 81    [dCJ_BR, ~, CT_BR] = propulsor_perf(d_BR, airplane.propulsion.right_blower, cbar, alt, V);
 82    [~, T_L, CT_CL] = propulsor_perf(d_TL, airplane.propulsion.left_cruiser, cbar, alt, V);
 83    [~, T_R, CT_CR] = propulsor_perf(d_TR, airplane.propulsion.right_cruiser, cbar, alt, V);
 84    
 85%	CL Calculations 
 86%	====================================
 87
 88    %Wing Lift Coefficient
 89    %C_L_wing
 90    %Currently this interpolator is only for 40deg flaps; any
 91    %flap changes are ignored.
 92    cl_left = funcCL(a_w_deg, dCJ_BL);
 93    cl_right = funcCL(a_w_deg, dCJ_BR);
 94    
 95    CLw = .9*(cl_left + cl_right)/2;
 96    
 97    %Tail Lift Coefficient
 98    %C_L_ht
 99    eta_h = 1; %Update with better method of estimating tail dynamic pressure ratio
100    eps = 2*CLw/(pi*e*AR); %May want to use M&S method here instead
101    a_h = (alphar+i_t-eps);
102    
103    eta = clat/(2*pi);
104    sweep_h = airplane.geometry.Htail.sweep;
105    CLah = (2*pi*AR)/(2+sqrt(4*(AR/eta)^2*(1+tan(sweep_h)^2)));
106    CLt = CLah*a_h*Sh/Sw*eta_h;
107    %Elevator and Pitch Rate effects
108    CLde = airplane.stability.CLde;
109    CLde_check = Sh/Sw*CLah*tau_h;
110    CLde = CLde_check; %Very different drom JVL; need to debug
111    CLq = airplane.stability.CLq;
112    CLq_check = -2*Vh*CLah;
113    CLq = CLq_check; %Very different drom JVL; need to debug
114    CL = CLw + CLah*a_h + CLq*x(3) + CLde*u(1);
115
116%	CX Calculations 
117%	====================================
118    cx_left = funcCX(a_w_deg, dCJ_BL);
119    cx_right = funcCX(a_w_deg, dCJ_BR);
120    
121    CXw = (cx_left+cx_right)/2;
122    CDi = CLw^2/(pi*AR*e+(CT_BL+CT_BR));
123    CDp = .03;  %Placeholder, this has little effect on the high-lift cases
124    
125    CX = CXw + CDi + CDp - CT_CL - CT_CR;
126
127%	Cm Calculations 
128%	====================================    
129    cm_left     = funcCM(a_w_deg, dCJ_BL);
130    cm_right    = funcCM(a_w_deg, dCJ_BR); 
131    
132    Cmw = (cm_left + cm_right)/2;
133    Cmh = -Vh*eta_h*CLah*a_h;
134    
135    Cmde = airplane.stability.Cmde;
136    Cmq = airplane.stability.Cmq;
137    
138    Cmaf = 0; %Negelect fuselage contributions for now
139    Cmf = Cmaf*alphar;
140    
141    Cm = CLw*(x_cg-x_acw)/cbar + Cmw + Cmh + Cmf + Cmde*u(1) + Cmq*x(3);
142    
143    	
144%	Current Lateral-Directional Characteristics
145%	===========================================
146%	Cl Calculations 
147%	==================================== 
148%	Rolling Moment Coefficient
149
150	%Cl      =	(ClBr*betar + CldRr*u(2)) + Clrr * x(7) + Clpr * x(6) ...
151    %            + (CldAr*u(3));% + CldASr*u(5));
152									% Total Rolling-Moment Coefficient
153    Cl = 0;
154
155%	CY Calculations 
156%	==================================== 
157%	Side-Force Coefficient
158	%CY	=	(CYBr*betar + CYdRr*u(2)) + (CYdAr*u(3));% + CYdASr*u(5));
159									% Total Side-Force Coefficient
160    CY = 0;
161%	Cn Calculations 
162%	====================================                                     
163%	Yawing Moment Coefficient
164	%Cn	=	(CnBr*betar + CndRr*u(2)) + Cnrr * x(7) + Cnpr * x(6) ...
165	%		+ (CndAr*u(3));% + CndASr*u(5));
166									% Total Yawing-Moment Coefficient
167    Cn = 0;
