1function [T, Vj] = get_motor_T(Pshaft, V, rho,propulsor)
 2%%%%%%%%%%%%%%%%%%%%%
 3%Computes motor thrust as a function of input power, motor size, and flight
 4%condition from actuator disk theory. 
 5%
 6%Inputs:    propulsor           Propulsor object, which contains the
 7%                               following attributes:
 8%               .eta_v          Prop viscous losses
 9%               .eta_add        Prop swirl losses
10%               .R               Prop radius, m
11%           Pshaft              Motor shaft power, W
12%           V                   Freestream velocity, m/s
13%           rho                 air density, kg/m^3
14%
15%Outputs:   T                   Net thrust produced by the disk in
16%                               isolation, N
17%           Vj                  Far downstream wake velocity, m/s
18
19eta_v = propulsor.eta_v;
20eta_add = propulsor.eta_add;
21R = propulsor.R;
22
23eta_p_i = .9;
24err = 1e6;
25thr = 1e-3;
26iter = 0;
27itermax = 100;
28
29
30
31while err > thr && iter < itermax
32    T = Pshaft*eta_p_i/V;
33    Tc = T/(rho*V^2*.5*pi*R^2);
34    eta_i = 2/(2+(sqrt(1+Tc)-1)/eta_add);
35    eta_p = eta_i*eta_v;
36    err = abs(eta_p_i - eta_p);
37    eta_p_i = eta_p;
38    iter = iter+1;
39end
40
41disp(err)
42disp(iter)
43
44w = V*(1/eta_i-1);
45Vj = V + 2*w;
