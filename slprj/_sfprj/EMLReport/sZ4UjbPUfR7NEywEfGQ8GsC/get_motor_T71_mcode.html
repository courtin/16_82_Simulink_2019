 2function [T, Vj] = get_motor_T(Pshaft, V, rho,propulsor)
 3%%%%%%%%%%%%%%%%%%%%%
 4%Computes motor thrust as a function of input power, motor size, and flight
 5%condition from actuator disk theory. 
 6%
 7%Inputs:    propulsor           Propulsor object, which contains the
 8%                               following attributes:
 9%               .eta_v          Prop viscous losses
10%               .eta_add        Prop swirl losses
11%               .R               Prop radius, m
12%           Pshaft              Motor shaft power, W
13%           V                   Freestream velocity, m/s
14%           rho                 air density, kg/m^3
15%
16%Outputs:   T                   Net thrust produced by the disk in
17%                               isolation, N
18%           Vj                  Far downstream wake velocity, m/s
19
20eta_v = propulsor.eta_v;
21eta_add = propulsor.eta_add;
22R = propulsor.R;
23
24eta_p_i = .9;
25err = 1e6;
26thr = 1e-3;
27iter = 0;
28itermax = 100;
29
30
31eta_i = 1;
32
33T=0;
34
35
36while err > thr && iter < itermax
37    T = Pshaft*eta_p_i/V;
38    Tc = T/(rho*V^2*.5*pi*R^2);
39    eta_i = 2/(2+(sqrt(1+Tc)-1)/eta_add);
40    eta_p = eta_i*eta_v;
41    err = abs(eta_p_i - eta_p);
42    eta_p_i = eta_p;
43    iter = iter+1;
44end
45
46w = V*(1/eta_i-1);
47Vj = V + 2*w;
