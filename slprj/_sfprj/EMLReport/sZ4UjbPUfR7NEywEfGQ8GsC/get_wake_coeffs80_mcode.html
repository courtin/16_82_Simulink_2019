1function [CQ, CJ, dCJ, CT, hj_c] = get_wake_coeffs(Vj, Vi, hd_c)
 2%%%%%%%%%%%%%%%%%
 3%Computes the dimensionless jet coefficients as a function of jet velocity,
 4%flight condition, and propeller radius, as well as the downstream jet height.
 5%
 6%Inputs:    r_prop          Propeller radius (same units as c)
 7%           Vj              Wake jet velocity (same units as V_inf)
 8%           Vi              Free-stream velocity (same units as V_j)
 9%           c               Wing chord (same units as r_prop)
10%
11%Outputs:   CQ              Jet mass flow coefficient
12%           CJ              Jet momentum coefficient
13%           dCJ             Net jet momentum coefficient
14%           CT              Thrust coefficient
15
16
17hj_c = hd_c/2*(1+Vi/Vj);
18CQ  = 1/2*(Vj/Vi+1)*hd_c;
19CJ  = 2*CQ*Vj/Vi;
20dCJ = 2*CQ*(Vj/Vi-Vi/Vj);
21CT  = CJ - 2*CQ;
22end
